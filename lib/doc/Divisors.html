def self.divcount (n)
        return nil if n < 0
        return 1 if n == 1
        divcount = 1
        Primes::factor(n).values.each {|n| divcount *= (n+1)}
        return divcount
end

def self.divisor_sigma (n, k)
        return self.divcount(n) if k == 0
        res = 0
        for i in self.divisors(n)
                res += i**k
        end
        return res
end

def self.divisors (n)
        factors = Primes::factor(n)
        ps = factors.keys.sort!
        return self._divisors(0, factors, ps).sort!.uniq
end
    euler_phi(n) click to toggle source

    Returns the valuer of phi(n), the Euler phi function; i.e. the number of integers in [1..n] comprime to n.

    Example

    >> Divisors::euler_phi(30)
=> 8

>> Divisors::euler_phi(37)
=> 36

    Algorithm

    n is first factored, then the result is computed as n * prod_{p} (1 - 1/p) where the product spans over all the prime factors of n

    # File divisors.rb, line 124
def self.euler_phi (n)
        return 0 if n < 1
        res = n
        Primes::factor(n).keys.each do |i|
                res *= i - 1
                res /= i
        end
        return res
end

def self.multiplicity (n, d)
        return 0 if n % d != 0
        m = n
        res = 0
        while m % d == 0
                m /= d
                res += 1
        end
        return res
end
