    class Primes::Sieve

    Class containing a list of all prime numbers up to a certain limit. All the methods that require such a list for a computation query the class using its accessors methods. The list is dynamically extended when needed.

def self.nthprime (n)
        if @primes[n] == nil
                high = n*(Math.log(n) + Math.log(Math.log(n))).to_i
                Sieve.extend(high)
                @limit = high
        end
        return @primes[n-1]
end

def self.primes_list (low = 1, high)
        low = low.to_i
        high = high.to_i
        if high > @limit
                Sieve.extend(high)
                @limit = high
        end
        l = @primes.index {|n| n >= low}
        h = @primes.rindex {|n| n <= high}
        return @primes[l..h]
end
